<script>

  // Passwörter

  const FRIEND_PASS = "geheim123";

  const ADMIN_PASS = "admin123";

 

  let userRole = null; // "friend" oder "admin"

  let today = new Date();

  let currentYear = today.getFullYear();

  let currentMonth = today.getMonth();

 

  // Daten aus localStorage

  let bookings = JSON.parse(localStorage.getItem("bookings")) || [];

  let blockedDays = JSON.parse(localStorage.getItem("blockedDays")) || [];

 

  // Auswahl für Buchung

  let selectingBooking = false;

  let bookingStart = null;

  let bookingEnd = null;

 

  // Elemente

  const loginBox = document.getElementById("login-box");

  const calendarApp = document.getElementById("calendar-app");

  const welcomeMsg = document.getElementById("welcome-msg");

  const monthYear = document.getElementById("month-year");

  const calendarEl = document.getElementById("calendar");

  const bookingInfo = document.getElementById("booking-info");

  const bookingDatesEl = document.getElementById("booking-dates");

  const bookingNameInput = document.getElementById("booking-name");

  const bookingError = document.getElementById("booking-error");

  const adminSection = document.getElementById("admin-section");

  const blockedDaysList = document.getElementById("blocked-days-list");

  const logoutBtn = document.getElementById("logout-btn");

 

  // Wochentage

  const dayNames = ["So", "Mo", "Di", "Mi", "Do", "Fr", "Sa"];

 

  function checkLogin() {

    const pass = document.getElementById("password").value.trim();

    if (pass === FRIEND_PASS) {

      userRole = "friend";

      afterLogin();

    } else if (pass === ADMIN_PASS) {

      userRole = "admin";

      afterLogin();

    } else {

      document.getElementById("error").textContent = "❌ Falsches Passwort!";

      return;

    }

  }

 

  function afterLogin() {

    loginBox.style.display = "none";

    calendarApp.style.display = "block";

    bookingError.textContent = "";

    bookingNameInput.value = "";

    bookingInfo.style.display = "none";

    blockedDaysList.innerHTML = "";

    bookingStart = null;

    bookingEnd = null;

    renderCalendar();

    if(userRole === "friend") {

      welcomeMsg.textContent = "🎉 Hallo Freunde! Buche hier deine Übernachtung.";

      adminSection.style.display = "none";

    } else {

      welcomeMsg.textContent = "🔧 Admin-Bereich: Sperrtage verwalten";

      adminSection.style.display = "block";

      renderBlockedDaysList();

    }

  }

 

  function renderCalendar() {

    calendarEl.innerHTML = "";

    monthYear.textContent = new Date(currentYear, currentMonth).toLocaleString("de-DE", {month:"long", year:"numeric"});

 

    // Wochentagsnamen

    for(let i=0; i<7; i++) {

      const d = document.createElement("div");

      d.className = "day-name";

      d.textContent = dayNames[i];

      calendarEl.appendChild(d);

    }

 

    // Erster Tag des Monats

    const firstDay = new Date(currentYear, currentMonth, 1);

    const startDay = firstDay.getDay(); // 0=So, 1=Mo...

 

    // Tage aus vorherigem Monat zum Ausgleich

    for(let i=0; i<startDay; i++) {

      const empty = document.createElement("div");

      empty.className = "day disabled";

      calendarEl.appendChild(empty);

    }

 

    // Anzahl Tage im Monat

    const daysInMonth = new Date(currentYear, currentMonth+1, 0).getDate();

 

    for(let day=1; day<=daysInMonth; day++) {

      const date = new Date(currentYear, currentMonth, day);

      const dayDiv = document.createElement("div");

      dayDiv.className = "day";

      dayDiv.textContent = day;

 

      const dateISO = date.toISOString().slice(0,10);

 

      // Prüfen ob Tag gebucht oder blockiert

      if (isDateBlocked(dateISO)) {

        dayDiv.classList.add("blocked");

      } else if (isDateBooked(dateISO)) {

        dayDiv.classList.add("booked");

      }

 

      // Admin und Freund haben unterschiedliche Klick-Funktionen

      dayDiv.addEventListener("click", () => {

        if(userRole === "admin") {

          toggleBlockedDay(dateISO);

        } else if(userRole === "friend") {

          friendSelectDate(dateISO, dayDiv);

        }

      });

 

      calendarEl.appendChild(dayDiv);

    }

  }

 

  function isDateBlocked(dateISO) {

    return blockedDays.includes(dateISO);

  }

 

  function isDateBooked(dateISO) {

    return bookings.some(b => {

      return dateISO >= b.start && dateISO <= b.end;

    });

  }

 

  // Admin-Funktion: Sperrtag setzen oder entfernen

  function toggleBlockedDay(dateISO) {

    const idx = blockedDays.indexOf(dateISO);

    if(idx >= 0) {

      blockedDays.splice(idx,1);

    } else {

      // Sperrtag darf nicht auf bereits gebuchten Tag gesetzt werden

      if(isDateBooked(dateISO)) {

        alert("Dieser Tag ist bereits gebucht und kann nicht gesperrt werden.");

        return;

      }

      blockedDays.push(dateISO);

    }

    saveBlockedDays();

    renderBlockedDaysList();

    renderCalendar();

  }

 

  function saveBlockedDays() {

    localStorage.setItem("blockedDays", JSON.stringify(blockedDays));

  }

 

  function renderBlockedDaysList() {

    blockedDaysList.innerHTML = "";

    blockedDays.forEach(d => {

      const span = document.createElement("span");

      span.textContent = d;

      span.title = "Klicke zum Entfernen";

      span.onclick = () => {

        toggleBlockedDay(d);

      };

      blockedDaysList.appendChild(span);

    });

  }

 

  // Freund-Funktion: Datumsauswahl Start/End

  function friendSelectDate(dateISO, dayDiv) {

    if(!selectingBooking) {

      if(isDateBlocked(dateISO)) {

        alert("Dieser Tag ist gesperrt. Bitte anderen Tag wählen.");

        return;

      }

      if(isDateBooked(dateISO)) {

        alert("Dieser Tag ist bereits gebucht. Bitte anderen Tag wählen.");

        return;

      }

      selectingBooking = true;

      bookingStart = dateISO;

      bookingEnd = null;

      clearSelectedDays();

      dayDiv.classList.add("selected");

      bookingInfo.style.display = "block";

      bookingDatesEl.textContent = `Startdatum: ${bookingStart}`;

      bookingNameInput.value = "";

      bookingError.textContent = "";

    } else {

      // Auswahl des Enddatums

      if(dateISO < bookingStart) {

        alert("Enddatum darf nicht vor Startdatum liegen.");

        return;

      }

      // Prüfen ob zwischen Start- und Enddatum geblockt oder gebucht

      const datesInRange = getDatesInRange(bookingStart, dateISO);

      for(const d of datesInRange) {

        if(isDateBlocked(d)) {

          alert("Zwischen Start- und Enddatum gibt es gesperrte Tage. Bitte andere Daten wählen.");

          return;

        }

        if(isDateBooked(d)) {

          alert("Zwischen Start- und Enddatum gibt es bereits gebuchte Tage. Bitte andere Daten wählen.");

          return;

        }

      }

      bookingEnd = dateISO;

      highlightSelectedRange(bookingStart, bookingEnd);

      bookingDatesEl.textContent = `Zeitraum: ${bookingStart} bis ${bookingEnd}`;

      bookingError.textContent = "";

    }

  }

 

  function clearSelectedDays() {

    document.querySelectorAll(".day.selected, .day.range").forEach(el => {

      el.classList.remove("selected", "range");

    });

  }

 

  function highlightSelectedRange(start, end) {

    clearSelectedDays();

    const days = document.querySelectorAll(".day");

    days.forEach(dayEl => {

      const dayNum = parseInt(dayEl.textContent);

      if(!dayNum) return;

      const date = new Date(currentYear, currentMonth, dayNum);

      const dateISO = date.toISOString().slice(0,10);

      if(dateISO === start || dateISO === end || (dateISO > start && dateISO < end)) {

        dayEl.classList.add("selected", "range");

      }

    });

  }

 

  function getDatesInRange(start, end) {

    const dates = [];

    let current = new Date(start);

    const endDate = new Date(end);

    while(current <= endDate) {

      dates.push(current.toISOString().slice(0,10));

      current.setDate(current.getDate() + 1);

    }

    return dates;

  }

 

  function saveBooking() {

    const name = bookingNameInput.value.trim();

    if(!bookingStart || !bookingEnd) {

      bookingError.textContent = "Bitte Start- und Enddatum auswählen.";

      return;

    }

    if(name.length < 2) {

      bookingError.textContent = "Bitte deinen Namen eingeben.";

      return;

    }

    // Noch mal sicherstellen, dass keine Überschneidungen bestehen

    const datesInRange = getDatesInRange(bookingStart, bookingEnd);

    for(const d of datesInRange) {

      if(isDateBlocked(d)) {

        bookingError.textContent = "Zwischen Start- und Enddatum gibt es gesperrte Tage.";

        return;

      }

      if(isDateBooked(d)) {

        bookingError.textContent = "Zwischen Start- und Enddatum gibt es bereits gebuchte Tage.";

        return;

      }

    }

    // Speichern

    bookings.push({name: name, start: bookingStart, end: bookingEnd});

    localStorage.setItem("bookings", JSON.stringify(bookings));

    bookingInfo.style.display = "none";

    selectingBooking = false;

    bookingStart = null;

    bookingEnd = null;

    bookingNameInput.value = "";

    bookingError.textContent = "";

    alert("Deine Übernachtung wurde eingetragen! Danke 😊");

    renderCalendar();

  }

 

  function cancelBooking() {

    bookingInfo.style.display = "none";

    selectingBooking = false;

    bookingStart = null;

    bookingEnd = null;

    bookingNameInput.value = "";

    bookingError.textContent = "";

    clearSelectedDays();

  }

 

  function prevMonth() {

    currentMonth--;

    if(currentMonth < 0) {

      currentMonth = 11;

      currentYear--;

    }

    renderCalendar();

  }

 

  function nextMonth() {

    currentMonth++;

    if(currentMonth > 11) {

      currentMonth = 0;

      currentYear++;

    }

    renderCalendar();

  }

 

  function logout() {

    userRole = null;

    loginBox.style.display = "block";

    calendarApp.style.display = "none";

    document.getElementById("password").value = "";

    document.getElementById("error").textContent = "";

  }

 

</script>
